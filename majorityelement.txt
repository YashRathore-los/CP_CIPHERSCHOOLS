//Mooreâ€™s Voting Algorithm
int Moore(std::vector<int> v) {
	int maj = v[0], counter = 1;
	for (int i = 1; i < v.size(); i++) {
		if (v[i] == maj) {
			counter++;
		}
		else {
			counter--;
		}
		if (counter == 0) {
			maj = v[i];
			counter = 1;
		}
	}
	return maj;
}

bool check(std::vector<int> v, int maj) {
	int cnt = 0;
	for (int i = 0; i < v.size(); i++) {
		if (v[i] == maj) {
			cnt++;
		}
		if (cnt > v.size() / 2) {
			return true;
		}
	}
	return false;
}
