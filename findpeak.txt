int solve(std::vector<int> v, int l, int r) {
	int mid = l + (r - l) / 2;
	if (mid == 0 && v[mid] >= v[mid + 1]) {
		return v[mid];
	}
	if (mid == v.size() - 1 && v[mid - 1]) {
		return v[mid];
	}
	if (v[mid] >= v[mid - 1] && v[mid] >= v[mid + 1]) {
		return v[mid];
	}
	else {
		if (mid > 0 && v[mid] < v[mid + 1]) {
			return solve(v, mid + 1, r);
		}
		else {
			return solve(v, l, mid - 1);
		}

	}
}
