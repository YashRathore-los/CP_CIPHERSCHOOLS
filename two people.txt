bool Naive(int x1, int x2, int v1, int v2) {
	if (x1 > x2 && v1 > v2) {
		return false;
	}
	if (x2 > x1 && v2 > v1) {
		return false;
	}
	if (x2 > x1) {
		swap(x2, x1);
		swap(v2, v1);
	}
	while (x1 >= x2) {
		if (x1 == x2) {
			return true;
		}
		x1 += v1;
		x2 += v2;
	}
	return false;
}

bool Efficient(int x1, int x2, int v1, int v2) {
	if (x1 > x2 && v1 > v2) {
		return false;
	}
	if (x2 > x1 && v2 > v1) {
		return false;
	}
	if (abs(x1 - x2) % abs(v1 - v2) == 0) {
		return true;
	}
	return false;
}
