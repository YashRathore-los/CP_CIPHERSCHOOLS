vector<int> solve(int arr[], int N)
{

	sort(arr, arr + N);
	vector<int> ans;

	int i = 0, j = N - 1;
	while (i < j) {
		ans.push_back(arr[j--]);
		ans.push_back(arr[i++]);
	}


	if (N % 2 != 0)
		ans.push_back(arr[i]);
	return ans;

}
