int solve(std::vector<int> v, int x, int l, int h) {
	if (l > h) {
		return -1;
	}
	int mid = (l + h) / 2;
	if (v[mid] == x) {
		return mid;
	}
	else if (v[mid] >= v[l]) {
		if (x <= v[mid] && x >= v[l]) {
			return solve(v, x, l, mid - 1);
		}
		else {
			return solve(v, x, mid + 1, h);
		}
	}
	else {
		if (x >= v[mid] && x <= v[h]) {
			return solve(v, x, mid + 1, h);
		}
		else {
			return solve(v, x, l, mid - 1);
		}
	}

}
